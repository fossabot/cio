261 static inline void cio_write_buffer_splice(struct cio_write_buffer *list, struct cio_write_buffer *head)
262 {
263  if (!cio_write_buffer_queue_empty(list)) {
264  struct cio_write_buffer *new_last = list->prev;
265  struct cio_write_buffer *new_first = list->next;
266  struct cio_write_buffer *last = head->prev;
267 
268  last->next = list->next;
269  new_first->prev = last;
270  new_last->next = head;
271  head->prev = new_last;
272 
273  head->data.head.q_len += list->data.head.q_len;
274  head->data.head.total_length += list->data.head.total_length;
275 
276  cio_write_buffer_head_init(list);
277  }
278 }
