86 static inline enum cio_error cio_read_buffer_init(struct cio_read_buffer *rb, void *data, size_t size)
87 {
88  if (cio_unlikely((rb == NULL) || (data == NULL) || (size == 0))) {
89  return CIO_INVALID_ARGUMENT;
90  }
91 
92  rb->data = data;
93  rb->end = (uint8_t *)data + size;
94  rb->fetch_ptr = data;
95  rb->add_ptr = data;
96 
97  return CIO_SUCCESS;
98 }
